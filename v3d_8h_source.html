65 // Compute a*X + Y -> Z
66 inline void v3d_axpy(const double a, const double *X, const double *Y, double *Z) {
67  Z[0] = a*X[0] + Y[0];
68  Z[1] = a*X[1] + Y[1];
69  Z[2] = a*X[2] + Y[2];

72 // normalize vector. Return "false and leave A unchanged if norm of A is
73 // less than min_norm or greater than max_norm
74 inline bool v3d_normalize(double *A, const double min_norm=1.0e-8, const double max_norm=1.0e8) {
75  double tval = v3d_norm(A);
76  if ( tval < min_norm || tval > max_norm)
77  return false;
78  else
79  v3d_scm(1.0/tval, A);
80  return true;
81 }

83 // Compute cross product of two vectors
84 inline void v3d_cross_product(const double *u, const double *v, double *X) {
85  X[0] = u[1]*v[2]-u[2]*v[1];
86  X[1] = -1.0*(u[0]*v[2]-u[2]*v[0]);
87  X[2] = u[0]*v[1]-u[1]*v[0];
88  return;
89 }
