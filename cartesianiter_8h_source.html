106 inline void RedundantCartesianIter::start()
107 {
108  if (l_==0)
109  done_ = 1;
110  else
111  done_ = 0;
112 
113  for (int i=0; i<l_; i++)
114  axis_[i] = 0;

117 inline void RedundantCartesianIter::next()
118 {
119  for (int i=0; i<l_; i++) {
120  if (axis_[i] == 2)
121  axis_[i] = 0;
122  else {
123  axis_[i]++;
124  return;
125  }
126  }
127  done_ = 1;
128 }
