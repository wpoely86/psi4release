95 void load(Archive &ar, psi::Matrix & t, unsigned int /*version*/)
96 {
97  std::string name;
98  psi::Dimension rows, cols;
99  int nirrep, symmetry;
100  ar & make_nvp("name", name);
101  ar & make_nvp("nirrep", nirrep);
102  ar & make_nvp("symmetry", symmetry);
103  ar & make_nvp("rowspi", rows);
104  ar & make_nvp("colspi", cols);
105 
106  t.init(rows, cols, name, symmetry);
107  for (int h=0; h < t.nirrep(); ++h) {
108  if (t.rowdim(h) != 0 && t.coldim(h))
109  ar & make_nvp("data", make_binary_object(t.pointer(h)[0], sizeof(double)*t.rowdim(h)*t.coldim(h)));
110  }
