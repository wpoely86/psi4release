153  static double a(const Vector3 &a1, const Vector3 &a2, const Vector3 &a3)
154  { Vector3 eBA(a2-a1), eBC(a2-a3); eBA.normalize(); eBC.normalize();
155  double costheta = eBA.dot(eBC);
156  if(costheta > 1.0 - CLEANUP_THRESH) costheta = 1.0;
157  if(costheta < CLEANUP_THRESH - 1.0) costheta = -1.0;
158  return acos(costheta); }
160  static double d(const Vector3 &a1, const Vector3 &a2, const Vector3 &a3, const Vector3 &a4)
161  { Vector3 eBA(a2-a1), eDC(a4-a3), eCB(a3-a2);
162  double CBNorm = eCB.norm();
163  Vector3 DCxCB(eDC.cross(eCB));
164  Vector3 CBxBA(eCB.cross(eBA));
165  return -atan2(CBNorm * eDC.dot(eCB.cross(eBA)), DCxCB.dot(CBxBA));}
