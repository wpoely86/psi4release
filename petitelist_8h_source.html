256  unsigned short dcr(unsigned short subgroup1, unsigned short subgroup2) const {
257  std::map<unsigned short,bool> uniqueCosets;
258  for(int g = 0; g < 8; ++g){
259  int coset = 0;
260  if(SKIP_THIS_OPERATOR(group_, g)) continue;
261  for(int mu = 0; mu < 8; ++mu){
262  if(SKIP_THIS_OPERATOR(subgroup1, mu)) continue;
263  for(int nu = 0; nu < 8; ++nu){
264  if(SKIP_THIS_OPERATOR(subgroup2, nu)) continue;
265  coset |= NUM_TO_OPERATOR_ID(mu^g^nu);
266  if(!NUM_TO_OPERATOR_ID(mu^g^nu)) coset |= SymmOps::ID;
267  }
268  }
269  uniqueCosets[coset] = 1;
270  }
271  std::map<unsigned short, bool>::const_iterator iter = uniqueCosets.begin();
272  std::map<unsigned short, bool>::const_iterator stop = uniqueCosets.end();
273  int rOperators = 0;
274  for(; iter != stop; ++iter){
275  int coset = iter->first;
276  for(int op = 1; op < 9; ++op){
277  if(SKIP_THIS_OPERATOR(coset, op)) continue;
278  rOperators |= (coset & SymmOps::ID ? SymmOps::E : NUM_TO_OPERATOR_ID(op));
279  break;
280  }
281  }
282  return rOperators;
283  }

300  std::vector<int> bits_to_operator_list(unsigned short list) {
301  std::vector<int> positions;
302  int position = 1;

304  positions.push_back(0);
305  for (int n=1; n<9; n++) {
